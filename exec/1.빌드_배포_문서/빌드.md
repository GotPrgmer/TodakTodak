pipeline {
    agent any
    tools{
        nodejs 'node16'
    }

    stages {
        stage('Checkout and Clone') {
            steps {
                git branch: 'develop', credentialsId: 'todaktodak', url: 'https://lab.ssafy.com/s08-final/S08P31B202'}
        }
        
        stage('Clone success'){
            steps {
                echo 'Clone success'
                sh 'ls -al'
            }
        }
        
        stage('React Build') {
            steps {
                dir('FrontEnd') {
                    sh 'npm i --save --legacy-peer-deps'
                    sh 'npm run build'
                }
            }
        }
        
        stage('React Build success'){
            steps {
                echo 'React build success'
                sh 'ls -al'
            }
